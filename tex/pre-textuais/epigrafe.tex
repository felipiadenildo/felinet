\begin{epigrafe}
   \vspace*{\fill}
   \begin{flushright}
      \textit{
         ``And God, please let the deer\\
         on the highway\\
         get some kind of heaven.\\
         Something with tall soft grass\\
         and sweet reunion.\\[8pt]
         Let the moths in porch lights\\
         go some place\\
         with a thousand suns,\\
         that taste like sugar\\
         and get swallowed whole.\\[8pt]
         May the mice\\
         in oil and glue\\
         have forever dry, warm fur\\
         and full bellies.\\[8pt]
         If I am killed\\
         for simply living,\\
         let death be kinder\\
         than man.''
      }
      \\[6pt]
      --- Althea Davis
   \end{flushright}
   \vspace*{\fill}
\end{epigrafe}

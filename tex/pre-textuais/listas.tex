\begin{simbolos}
  \item[$\mu\pm\sigma$] Média e desvio padrão sobre sementes aleatórias
  \item[$N$] Tamanho do catálogo (galeria) em protocolos de re-ID
  \item[$K$] Posto no \textit{ranking} de hipóteses de re-ID (Top-$K$)
  \item[$\tau$] Limiar de rejeição em regime \textit{open-set}
  \item[Top-1, Top-5, Top-10] Fração de consultas com identidade correta entre as $K$ primeiras hipóteses
  \item[mAP@10] Precisão média avaliada até o décimo posto do \textit{ranking}
  \item[TPR@FPR=1\%] Taxa de verdadeiros positivos quando a taxa de falsos positivos é fixada em 1\%
  \item[IoU] Intersecção sobre união entre \textit{bounding boxes}
\end{simbolos}
